import js import sys import asyncio import math import random class GameState: def __init__(self, AI_hands=[1, 1], MY_hands=[1, 1], player_turn="MY"): self.AI_hands = AI_hands.copy() self.MY_hands = MY_hands.copy() self.player_turn = player_turn def copy(self): return GameState( AI_hands=self.AI_hands.copy(), MY_hands=self.MY_hands.copy(), player_turn=self.player_turn ) def __str__(self): return f"AI Hands: {self.AI_hands}, ME Hands: {self.MY_hands} , played : {self.player_turn}" class Move: def __init__(self, move_type, source_hand=None, target_hand=None, amount=None): self.type = move_type self.source_hand = source_hand self.target_hand = target_hand self.amount = amount def __str__(self): if self.type == "attack": return f"Attack from {self.source_hand} to {self.target_hand}" elif self.type == "split": return f"Split: Move {self.amount} to hand {0 if self.source_hand is None else self.source_hand}" class Node: def __init__(self, state, parent=None, move=None): self.state = state self.parent = parent self.children = [] self.visits = 0 self.wins = 0 self.move = move self.untried_moves = get_valid_moves(state) class ChopsticksMCTS: def __init__(self, exploration_weight=2.0): self.exploration_weight = exploration_weight def search(self, initial_state, num_simulations=1000): root = Node(initial_state) for _ in range(num_simulations): node = self._select(root) result = self._simulate(node.state) self._backpropagate(node, result) # Return the best move return self._get_best_move(root) def _select(self, node): while not is_terminal(node.state): if node.untried_moves: return self._expand(node) else: node = self._uct_select(node) return node def _expand(self, node): move = node.untried_moves.pop() next_state = apply_move(node.state, move) child_node = Node(next_state, parent=node, move=move) node.children.append(child_node) return child_node def _simulate(self, state): current_state = state.copy() while not is_terminal(current_state): moves = get_valid_moves(current_state) move = random.choice(moves) # print(move) current_state = apply_move(current_state, move) #print(current_state) return get_winner(current_state) def _backpropagate(self, node, result): while node: node.visits += 1 if result == node.state.player_turn: node.wins += 1 node = node.parent def _uct_select(self, node): return max( node.children, key=lambda c: c.wins / c.visits + self.exploration_weight * math.sqrt(math.log(node.visits) / c.visits) ) def _get_best_move(self, root): best_child = max(root.children, key=lambda c: c.visits) return best_child.move def get_valid_moves(state): moves = [] if state.player_turn == "MY": #
